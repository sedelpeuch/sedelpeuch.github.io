  GLOB_RECURSE
  source_files
  src/auxiliaire.c
  src/auxilaire.h
  src/global.c
  src/global.h
  src/good.c
  src/good.h
  src/queue.c
  src/queue.h
  src/stockex.c
  src/stockex.h
  )

  project
  ${source_files}
  src/loop.c
  src/loop.h
  )

  test
  ${source_files}
  tst/tst.c
  )
